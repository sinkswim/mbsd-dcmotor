274/*========================================================================*



275 * Function: SetSITErrorMessage



276 *



277 * Abstract:



278 *	Sets and prints the system's error or warning message to stderr. To clear the last message, you must first set ErrMsg to NULL.



279 *



280 * Parameters:



281 *	ErrMsg : error or warning string. First set to NULL to clear the last error message



282 *	isError : if true, the SITglobals.stopExecutionFlag is enabled and the message is reported as a Fatal Error.



283 *



284 *========================================================================*/



285void SetSITErrorMessage(const char *ErrMsg, int32_t isError)



286{



287  /* If a fatal error and one hasn't yet occurred */



288  if (isError && (SITglobals.stopExecutionFlag == 0)) {



289    /* Set the stop flag and forcefully set the error message by setting internal variable to NULL.*/



290    SITglobals.stopExecutionFlag = 1;



291    SITglobals.errmsg = NULL;



292  }



293



294  /* Only set the error message if it hasn't been set before. */



295  if (SITglobals.errmsg == NULL) {



296    SITglobals.errmsg = ErrMsg;



297  }



298



299  /* Print the error\warning message */



300  if (SITglobals.errmsg != NULL) {



301    if (isError) {



302      (void)fprintf(stderr,"VeriStand Error: %s\n", SITglobals.errmsg);



303    } else {



304      (void)fprintf(stderr,"VeriStand Warning: %s\n", SITglobals.errmsg);



305    }



306  }



307}

312/*========================================================================*



313 * Function: rtOneStep



314 *



315 * Abstract:



316 *	Perform one step of the model. This function is modeled such that



317 *	it could be called from an interrupt service routine (ISR) with minor



318 *	modifications.



319 *



320 * Returns:



321 *     NI_OK if no error.



322 *========================================================================*/



323static int32_t rt_OneStep(RT_MODEL *S, double *inData)



324{



325  real_T tnext = 0;



326  int_T *sampleHit = rtmGetSampleHitPtr(S);



327



328  /***********************************************



329   * Check and see if base step time is too fast *



330   ***********************************************/



331  if (SITglobals.isrOverrun++) {



332    SITglobals.stopExecutionFlag = 1;



333    return -1;



334  }



335



336  /***********************************************



337   * Check and see if error status has been set  *



338   ***********************************************/



339  if (rtmGetErrorStatus(S) != NULL) {



340    SITglobals.stopExecutionFlag = 1;



341    return -1;



342  }



343



344  /* enable interrupts here */



345  SetExternalInputs(inData, sampleHit);



346  tnext = rt_SimGetNextSampleHit();



347  rtsiSetSolverStopTime(rtmGetRTWSolverInfo(S),tnext);



348  MdlOutputs(0);



349  SITglobals.isrOverrun--;



350  return NI_OK;



351}                                      /* end rtOneStep */



352

356#else



357#define FIRST_TID                      0



358#endif

360/*========================================================================*



361 * Function: SCHEDULER



362 *



363 * Abstract:



364 *	Computes SampleHits for tasks in the model.



365 *



366 * Returns:



367 *	NI_OK if no error.



368 *========================================================================*/



369static int32_t SCHEDULER(RT_MODEL *S)



370{



371  int_T i = 0;



372  real_T tnext = 0;



373  int_T *sampleHit = rtmGetSampleHitPtr(S);



374



375  /***********************************************



376   * Check and see if base step time is too fast *



377   ***********************************************/



378  if (SITglobals.isrOverrun++) {



379    SITglobals.stopExecutionFlag = 1;



380    return -1;



381  }



382



383  /***********************************************



384   * Check and see if error status has been set  *



385   ***********************************************/



386  if (rtmGetErrorStatus(S) != NULL) {



387    SITglobals.stopExecutionFlag = 1;



388    return -1;



389  }



390



391  /* compute next cycle hits */



392  tnext = rt_SimUpdateDiscreteEvents(rtmGetNumSampleTimes(S),



393    rtmGetTimingData(S),



394    rtmGetSampleHitPtr(S),



395    rtmGetPerTaskSampleHitsPtr(S));



396  rtsiSetSolverStopTime(rtmGetRTWSolverInfo(S),tnext);



397



398  /*********************************************



399   * Step the model for any other sample times *



400   *********************************************/



401  for (i=1+FIRST_TID; i<NUMST; i++) {



402    if (sampleHit[i]) {



403      /*	if sampleHit, mark the overrunFlag : if the flag is already marked,



404         the task has not finished its run from last sampleHit.



405         Marking the flag in scheduler instead of in the task:



406         To detect the case of overrun when the task cannot even start



407         execution even though it is scheduled



408       */



409      if (SITglobals.overrunFlags[i] > 0) {



410        // check for overrun



411        printf("overrun in task %d: %d\n", i, SITglobals.overrunFlags[i]);



412        SITglobals.overrunCount[i]++;  // update counter



413        if (SITglobals.haltonOverrun) {



414          SetSITErrorMessage("Task Overran", 1);



415        }



416      } else {



417        SITglobals.overrunFlags[i] = 1;



418      }



419    }



420  }



421



422  SITglobals.isrOverrun--;



423  return NI_OK;



424}                                      /* end SCHEDULER */

426/*========================================================================*



427 * Function: NIRT_ScheduleTasks2



428 *



429 * Abstract:



430 *	Determines the sub tasks to execute next. We do not run the other tasks here, they are



431 *	executed separately through NIRT_TaskTakeOneStep)



432 *



433 * Output Parameters:



434 *	dispatchtasks	: Array size of NUMST.  This defines the next task to execute. dispatchtasks[0] is the base task.



435 *	isFirstSchedule	: true if executing the first time step



436 *



437 * Returns:



438 *	NI_OK if no error.



439 *========================================================================*/



440DLL_EXPORT int32_t NIRT_ScheduleTasks2(int32_t *dispatchtasks, char



441  *isFirstSchedule)



442{



443



444#ifndef MULTITASKING



445



446  SetSITErrorMessage("Called ScheduleTasks2() for a single-rate model.  Use Schedule() instead.",



447                     1);



448  return NI_ERROR;



449



450#else



451



452  int32_t tid = 0;



453  int32_t retVal = 0;



454  int_T *sampleHits = rtmGetSampleHitPtr(S);



455  retVal = SCHEDULER(S);



456  if (retVal == 0) {



457    /* Update with list of scheduled tasks.*/



458    for (tid=0; tid<NUMST; tid++) {



459      dispatchtasks[tid] = sampleHits[tid + TID01EQ];



460    }



461  }



462



463  *isFirstSchedule = SITglobals.isFirstTimeStep;



464  return retVal;



465



466#endif



467



468}

470/*========================================================================*



471 * Function: rt_OneStepTask



472 *



473 * Abstract:



474 *	Takes one step for the task specified by tid



475 *



476 * Input Parameters:



477 *	S		: Pointer to model information



478 *	tid		: Task ID to step



479 *	inData	: Pointer to inport data. Ignored unless it is the FIRST_TID (aka base rate).



480 *



481 * Returns:



482 *	1 if this is the last task to execute for a time step



483 *	0 if other tasks still need to execute



484 *	-1 if an error occurred



485 *========================================================================*/



486static int32_t rt_OneStepTask(RT_MODEL *S, int32_t tid, double *inData)



487{



488  if (tid == FIRST_TID) {



489    /* Return if there has been a previous error */



490    if (SITglobals.stopExecutionFlag) {



491      return -1;



492    }



493



494    /*******************************************



495     * Step the model for the base sample time *



496     *******************************************/



497    if ((inData == NULL) && (NumInputPorts() > 0)) {



498      /* For baserate (FIRST_TID) we need inData, but only set error message if it hasn't been set yet */



499      SetSITErrorMessage("External Input data array is NULL for base rate task",



500                         1);



501      return -1;



502    }



503



504    /* MdlUpdate is not executed on the first iteration. */



505



506    /* pull inport data in from caller */



507    SetExternalInputs(inData, rtmGetSampleHitPtr(S));



508    MdlOutputs(FIRST_TID);



509  } else {



510    /*********************************************



511     * Step the model for any other sample times *



512     *********************************************/



513    if ((tid < (FIRST_TID+1)) || (tid >= NUMST) ) {



514      return -1;



515    }



516



517    MdlOutputs(tid);



518    MdlUpdate(tid);



519    rt_SimUpdateDiscreteTaskTime(rtmGetTPtr(S), rtmGetTimingData(S),tid);



520



521    /* Indicate task complete */



522    SITglobals.overrunFlags[tid]--;



523  }                                    // of tid==FIRST_TID



524



525  return NI_OK;



526}                                      /* end rtOneStep */

555/*========================================================================*



556 * Function: NIRT_InitializeModel



557 *



558 * Abstract:



559 *	Initializes internal structures and prepares model for execution.



560 *



561 * Input Parameters:



562 *	finaltime	: the final time until which the model should run.



563 *



564 * Output Parameters:



565 *	outTimeStep	: the base time step of the model scheduler.



566 *	num_in		: number of external inputs



567 *	num_out		: number of external outputs



568 *	num_tasks	: number of Tasks (for multirate models only).



569 *



570 * Returns:



571 *	NI_OK if no error



572 *========================================================================*/



573DLL_EXPORT int32_t NIRT_InitializeModel (double finaltime, double *outTimeStep,



574  int32_t *outNumInports, int32_t *outNumOutports, int32_t *numTasks)



575{



576  const char *status = NULL;



577



578  //Make sure SITglobals is reinitialized even if DLL is not reloaded



579  (void)memset(&SITglobals, 0, sizeof(SITglobals));



580  (void)memset(&SITexportglobals, 0, sizeof(SITexportglobals));



581  rt_InitInfAndNaN(sizeof(real_T));



582  NI_InitializeParamStruct();



583  S = MODEL();



584  if (rtmGetErrorStatus(S) != NULL) {



585    (void)fprintf(stderr,"Error during model registration: %s\n",



586                  rtmGetErrorStatus(S));



587    return 5;



588  }



589



590  if ((finaltime >= 0.0) || (finaltime == RUN_FOREVER)) {



591    rtmSetTFinal(S,finaltime);



592  }



593



594  MdlInitializeSizes();



595  MdlInitializeSampleTimes();



596  status = rt_SimInitTimingEngine(rtmGetNumSampleTimes(S),



597    rtmGetStepSize(S),



598    rtmGetSampleTimePtr(S),



599    rtmGetOffsetTimePtr(S),



600    rtmGetSampleHitPtr(S),



601    rtmGetSampleTimeTaskIDPtr(S),



602    rtmGetTStart(S),



603    &rtmGetSimTimeStep(S),



604    &rtmGetTimingData(S));



605  if (status != NULL) {



606    (void)fprintf(stderr, "Failed to initialize sample time engine: %s\n",



607                  status);



608    exit(EXIT_FAILURE);



609  }



610



611  rt_CreateIntegrationData(S);



612  SITglobals.errmsg = rt_StartDataLogging(rtmGetRTWLogInfo(S),



613    rtmGetTFinal(S),



614    rtmGetStepSize(S),



615    &rtmGetErrorStatus(S));



616  if (SITglobals.errmsg != NULL) {



617    (void)fprintf(stderr,"Error starting data logging: %s\n",SITglobals.errmsg);



618    return(3);



619  }



620



621  NI_InitParamDoubleBuf();



622  if (rtmGetErrorStatus(S) != NULL) {



623    SITglobals.stopExecutionFlag = 1;



624  }



625



626  NIRT_GetModelSpec(NULL, 0, outTimeStep, outNumInports, outNumOutports,



627                    numTasks);



628  InitializeCriticalSection(&SITglobals.taskCriticalSection);



629  SITexportglobals.flipCriticalSection = CreateSemaphore(NULL, 1, 1, NULL);



630  if (SITexportglobals.flipCriticalSection == NULL) {



631    SetSITErrorMessage("Failed to create semaphore.", 1);



632  }



633



634  return NI_OK;



635}                                      // end of InitializeModel()

669/*========================================================================*



670 * Function: NIRT_ModelUpdate



671 *



672 * Abstract:



673 *	Updates the internal state of the model.



674 *



675 *	Each call to Schedule() or ScheduleTasks() must be followed by a call to NIRT_ModelUpdate before



676 *	the scheduling call can be made again. Decrement our count for entries into the critical section,



677 *	signalling the it is no longer safe to probeUpdate states. Leave critical section.



678 *



679 * Returns:



680 *	NI_OK if no error



681 *========================================================================*/



682DLL_EXPORT int32_t NIRT_ModelUpdate(void)



683{



684  if (SITexportglobals.inCriticalSection) {



685    SITexportglobals.inCriticalSection--;



686    MdlUpdate(FIRST_TID);



687



688#if defined(MULTITASKING)



689



690    if (rtmGetSampleTime(S,0) == CONTINUOUS_SAMPLE_TIME) {



691      rt_UpdateContinuousStates(S);



692    } else {



693      rt_SimUpdateDiscreteTaskTime(rtmGetTPtr(S), rtmGetTimingData(S), 0);



694    }



695



696#if FIRST_TID == 1



697



698    rt_SimUpdateDiscreteTaskTime(rtmGetTPtr(S), rtmGetTimingData(S),1);



699



700#endif



701



702#else



703



704    rt_SimUpdateDiscreteTaskSampleHits(rtmGetNumSampleTimes(S), rtmGetTimingData



705      (S), rtmGetSampleHitPtr(S), rtmGetTPtr(S));



706    if (rtmGetSampleTime(S,0) == CONTINUOUS_SAMPLE_TIME) {



707      rt_UpdateContinuousStates(S);



708    }



709



710#endif



711



712    ReleaseSemaphore(SITexportglobals.flipCriticalSection,1,NULL);



713  }



714



715  return SITexportglobals.inCriticalSection;



716}

747  int32_t lastStep = 1;



748  int_T *sampleHits = rtmGetSampleHitPtr(S);



749



750  // donot increment the time..



751  if (outTime != NULL) {



752    *outTime = rtmGetT(S);



753  }



754



755  /* Critical section is used to prevent the background loop from switching parameters



756     sets while the baserate is running.  On a single core system this can not happen, but



757     on a multi-core system the baserate and background can be executing at the same time. */



758  WaitForSingleObject(SITexportglobals.flipCriticalSection, INFINITE);



759  if (SITexportglobals.inCriticalSection > 0) {



760    SetSITErrorMessage("Each call to StepBaseTask() MUST be followed by a call to ModelUpdate() before StepBaseTask() is called again.",



761                       1);



762    ReleaseSemaphore(SITexportglobals.flipCriticalSection, 1, NULL);



763    return 1;



764  }



765



766  /* For each task other than the base rate copy parameters if there are new ones.



767     Since the task is about the be scheduled it should not be executing at the moment



768     so copying the parameters will be safe.



769   */



770# ifdef NI_SIT_PARAMS



771



772  for (i = 1; i < NUMST - TID01EQ; i++) {



773    if (sampleHits[i + TID01EQ] && (SITexportglobals.copyTaskBitfield & (1 << i)))



774    {



775      memcpy(param_lookup[i+TID01EQ][1-READSIDE], param_lookup[0][READSIDE],



776             sizeof(NI_PARAM_TYPEDEF));



777      SITexportglobals.copyTaskBitfield &= ~(1 << i);



778    }



779  }



780



781# endif



782



783  //If stopExecutionFlag is true we have had an error



784  if (!SITglobals.stopExecutionFlag) {



785    if ((rtmGetTFinal(S) == RUN_FOREVER) || ((rtmGetTFinal(S) - rtmGetT(S)) >



786         (rtmGetT(S) *DBL_EPSILON))) {



787      rt_OneStepTask(S,FIRST_TID, inData);



788      lastStep = rtmGetStopRequested(S);



789    } else if (!rtmGetStopRequested(S)) {// take last step



790      rt_OneStepTask(S,FIRST_TID, inData);



791    }



792



793    /* Update outports */



794    SetExternalOutputs(outData, sampleHits);



795  }



796



797  SITexportglobals.inCriticalSection++;



798



799  /* Reset first run flag */



800  SITglobals.isFirstTimeStep = FALSE;



801  return SITglobals.stopExecutionFlag | lastStep;



802



803#endif



804



805}

807/*========================================================================*



808 * Function: NIRT_Schedule



809 *



810 * Abstract:



811 *	Scheduler behaves traditionally for singletasking (Get Inputs, Take Step, Post Outputs).



812 *	For multitasking, the outputs are delayed by one time step.



813 *



814 *	This scheduler ignores dispatchtasks input as all tasks are dispatched based on their sample



815 *	rate information alone. This is done externally by timed loops.



816 *



817 * Input Parameters:



818 *	inData	: Points to an array of elements sent to the model inports



819 *



820 * Output Parameters:



821 *	outData	: Points to an array of elements sent to the model outports



822 *	outTime	: Used to return the current model time.  Ignored if it is NULL



823 *



824 * Returns:



825 *	NI_OK if no error



826 *========================================================================*/



827DLL_EXPORT int32_t NIRT_Schedule(double *inData, double *outData, double



828  *outTime, int32_t *dispatchtasks)



829{



830



831#ifdef MULTITASKING



832



833  SetSITErrorMessage("Called Schedule() for a multi-rate model.  Use ScheduleTasks2() instead.",



834                     1);



835  return NI_ERROR;



836



837#else



838



839  int32_t lastStep = 1;



840  if (outTime != NULL) {



841    *outTime = rtmGetT(S);



842  }



843



844  WaitForSingleObject(SITexportglobals.flipCriticalSection, INFINITE);



845  if (SITexportglobals.inCriticalSection > 0) {



846    SetSITErrorMessage("Each call to Schedule() MUST be followed by a call to ModelUpdate() before Schedule() is called again.",



847                       1);



848    ReleaseSemaphore(SITexportglobals.flipCriticalSection, 1, NULL);



849    return 1;



850  }



851



852  if (!SITglobals.stopExecutionFlag && ((rtmGetTFinal(S) == RUN_FOREVER) ||



853       ((rtmGetTFinal(S) - rtmGetT(S)) > (rtmGetT(S) * DBL_EPSILON)))) {



854    rt_OneStep(S, inData);



855    lastStep = rtmGetStopRequested(S);



856  } else if (!SITglobals.stopExecutionFlag && !rtmGetStopRequested(S)) {



857    // take last step



858    rt_OneStep(S, inData);



859  }



860



861  SetExternalOutputs(outData, rtmGetSampleHitPtr(S));// push outport data out to LabVIEW



862  SITexportglobals.inCriticalSection++;



863  return lastStep;



864



865#endif



866



867}

980 * Input/Output Parameters:

987 *	NI_OK if no error

992  const char *sysmsg = NULL;

1010
