726 *	Otherwise, lookup parameter by index, and return the information.

732 *	bnlen		: length of buffer blkname (in), length of output blkname (out)



733 *	snlen		: length of buffer signame (in), length of output signame (out)

739 *	datatype	: same as with parameters. Currently assuming all data to be double.

746DLL_EXPORT int32_t NIRT_GetSignalSpec(int32_t* sidx, char* ID, int32_t* ID_len,



747  char* blkname, int32_t *bnlen,



748  int32_t *portnum, char* signame, int32_t *snlen, int32_t *dattype, int32_t*



749  dims, int32_t* numdim)



750{



751  int32_t sigidx = *sidx;



752  int32_t i = 0;



753  char *addr = NULL;



754  char *IDblk = 0;



755  int32_t IDport = 0;



756  if (sigidx < 0) {



757    /* check if ID has been specified. */



758    if ((ID != NULL) && (ID_len != NULL) && (*ID_len > 0)) {



759      /* parse ID into blkname and port */



760      addr = strrchr(ID, ':');



761      if (addr == NULL) {



762        return NI_SigListSize;



763      }



764



765      /* Calculate the char offset into the string after the port */



766      i = addr - ID;



767



768      /* malformed ID */



769      if (i<=0) {



770        return NI_SigListSize;



771      }



772



773      IDblk = ID;



774      IDport = atoi(ID+i+1);



775



776      /* lookup the table for matching ID */



777      for (i=0; i<NI_SigListSize; i++) {



778        /* 11 to accomodate ':','/', port number and null character */



779        char *tempID = (char *)calloc(strlen(NI_SigList[i].blockname)+strlen



780          (NI_SigList[i].signalname) + 11, sizeof(char));



781        if (strlen(NI_SigList[i].signalname)>0) {



782          sprintf(tempID,"%s:%d%s%s",NI_SigList[i].blockname,NI_SigList[i].



783                  portno+1,"/",NI_SigList[i].signalname);



784        } else {



785          sprintf(tempID,"%s:%d",NI_SigList[i].blockname,NI_SigList[i].portno+1);



786        }



787



788        if (!strcmp(IDblk,tempID) && IDport==(NI_SigList[i].portno+1)) {



789          break;



790        }



791



792        free(tempID);



793      }



794



795      if (i < NI_SigListSize) {



796        sigidx = *sidx = i;



797      } else {



798        return NI_SigListSize;



799      }



800    } else {



801      // no index or ID specified.



802      return NI_SigListSize;



803    }



804  }



805



806  if (sigidx>=0 && sigidx<NI_SigListSize) {



807    if (ID != NULL) {



808      // no need for return string to be null terminated!



809      /* 11 to accomodate ':','/', port number and null character */



810      char *tempID = (char *)calloc(strlen(NI_SigList[sigidx].blockname)+strlen



811        (NI_SigList[sigidx].signalname)+ 11, sizeof(char));



812      if (strlen(NI_SigList[sigidx].signalname)>0) {



813        sprintf(tempID,"%s:%d%s%s",NI_SigList[sigidx].blockname,



814                NI_SigList[sigidx].portno+1,"/",NI_SigList[sigidx].signalname);



815      } else {



816        sprintf(tempID,"%s:%d",NI_SigList[sigidx].blockname,NI_SigList[sigidx].



817                portno+1);



818      }



819



820      if ((int32_t)strlen(tempID)<*ID_len) {



821        *ID_len = strlen(tempID);



822      }



823



824      strncpy(ID, tempID, *ID_len);



825      free(tempID);



826    }



827



828    if (blkname != NULL) {



829      // no need for return string to be null terminated!



830      if ((int32_t)strlen(NI_SigList[sigidx].blockname)<*bnlen) {



831        *bnlen = strlen(NI_SigList[sigidx].blockname);



832      }



833



834      strncpy(blkname, NI_SigList[sigidx].blockname, *bnlen);



835    }



836



837    if (signame != NULL) {



838      // no need for return string to be null terminated!



839      if ((int32_t)strlen(NI_SigList[sigidx].signalname)<*snlen) {



840        *snlen = strlen(NI_SigList[sigidx].signalname);



841      }



842



843      strncpy(signame, NI_SigList[sigidx].signalname, *snlen);



844    }



845



846    if (portnum != NULL) {



847      *portnum = NI_SigList[sigidx].portno;



848    }



849



850    if (dattype != NULL) {



851      *dattype = NI_SigList[sigidx].datatype;



852    }



853



854    if (numdim != NULL) {



855      if (*numdim == -1) {



856        *numdim = NI_SigList[sigidx].numofdims;



857      } else if (dims != NULL) {



858        for (i=0;i < *numdim; i++) {



859          dims[i] = NI_SigDimList[NI_SigList[sigidx].dimListOffset +i];



860        }



861      }



862    }



863



864    return NI_OK;



865  }



866



867  return NI_SigListSize;



868}

1083/*========================================================================*



1084 * Function: NIRT_ProbeSignals



1085 *



1086 * Abstract:



1087 *	returns the latest signal values.



1088 *



1089 * Input Parameters:



1090 *	sigindices	: list of signal indices to be probed.



1091 *	numsigs		: length of sigindices array.



1092 *



1093 *		NOTE: (Implementation detail) the sigindices array that is passed in is delimited by a value of -1.



1094 *		Thus the # of valid indices in the sigindices table is min(numsigs, index of value -1) - 1.



1095 *		Reason for subtracting 1, the first index in the array is used for bookkeeping and should be copied



1096 *		into the 0th index in the signal values buffer.



1097 *		Also, the 1st index in the signal values buffer should contain the current timestamp. Hence valid



1098 *		signal data should be filled in starting from index 2. Any non-scalar signals should be added to the



1099 *		buffer in row-order.



1100 *



1101 * Input/Output Parameters



1102 *	num			: (in) length of sigvalues buffer (out) number of values returned in the buffer



1103 *



1104 * Output Parameters:



1105 *	value		: array of signal values



1106 *



1107 * Returns:



1108 *	NI_OK if no error



1109 *========================================================================*/



1110DLL_EXPORT int32_t NIRT_ProbeSignals(int32_t* sigindices, int32_t numsigs,



1111  double* value, int32_t* len)



1112{



1113  int32_t i = 0;



1114  int32_t idx = 0;



1115  int32_t count = 0;



1116  if (!SITexportglobals.inCriticalSection) {



1117    SetSITErrorMessage("SignalProbe should only be called between ScheduleTasks and PostOutputs",



1118                       1);



1119  }



1120



1121  /* Get the index to the first signal */



1122  if ((*len > 1) && (numsigs > 0)) {



1123    value[count++] = sigindices[0];



1124    value[count++] = 0;



1125  }



1126



1127  /* Get the second and other signals */



1128  for (i = 1; (i < numsigs) && (count < *len); i++) {



1129    idx = sigindices[i];



1130    if (idx < 0) {



1131      break;



1132    }



1133



1134    if (idx < NI_SigListSize) {



1135      if (NI_SigList[idx].baseaddr == VIRTUAL_SIG) {



1136        int32_t vidx = NI_VirtualBlockSources[NI_SigList[idx].addr];



1137        NI_ProbeOneSignal(vidx, value, *len, &count, 0, -1);



1138      } else {



1139        NI_ProbeOneSignal(idx, value, *len, &count, 0, -1);



1140      }



1141    }



1142  }



1143



1144  *len = count;



1145  return count;



1146}
